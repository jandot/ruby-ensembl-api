      def misc_features(code)
        answer = Array.new
        if code.nil?
          self.seq_region.misc_features.each do |mf|
            if mf.seq_region_start > self.start and mf.seq_region_end < self.stop
              answer.push(mf)
            end
          end
        else
          self.seq_region.misc_features.each do |mf|
            if mf.misc_sets[0].code == code
              if mf.seq_region_start > self.start and mf.seq_region_end < self.stop
                answer.push(mf)
              end
            end
          end
        end
        return answer
      end
