      def to_mappings
        output = Array.new
        self.marker_features.each do |mf|
          output.push(mf.seq_region.coord_system.name + "\t" + mf.seq_region.coord_system.version + "\t" + mf.seq_region.name + "\t" + mf.seq_region_start.to_s + "\t" + mf.seq_region_end.to_s)
        end
        return output.join("\n")

      end
